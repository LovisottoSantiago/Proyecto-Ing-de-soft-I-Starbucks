-- 3. Crear trigger para calcular autom√°ticamente el PrecioUnitario
CREATE TRIGGER trg_CalcularPrecioUnitario
ON Producto
AFTER INSERT, UPDATE
AS
BEGIN
    SET NOCOUNT ON;

    UPDATE p
    SET p.PrecioUnitario = p.PrecioCosto * 2.2
    FROM Producto p
    INNER JOIN inserted i ON p.ID_Producto = i.ID_Producto;
END;
